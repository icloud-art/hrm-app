create database hrm_db;

//使用数据库
use hrm_db;

//创建dept_inf表
create table dept_inf(id int(11) not null auto_increment,name varchar(50) not null,
remark varchar(300) default null,primary key (id)) engine = innodb auto_increment=10 default charset=utf8;

//创建job_inf表
create table job_inf(id int(11) not null auto_increment,name varchar(50) not null,
remark varchar(300) default null,primary key(id)) engine=innodb auto_increment=10 default charset=utf8;

//创建user_inf表
create table user_inf(id int(11) not null auto_increment,loginname varchar(20) not null,
password varchar(16) not null,status int (11) not null default '1',
createdate timestamp not null default current_timestamp,username varchar(20) default null,
primary key (id))engine=innodb auto_increment=2 default charset=utf8;

//创建notice_inf表
create table notice_inf(id int(11) not null auto_increment,title varchar(50) not null,
content text not null,create_date timestamp not null default current_timestamp,user_id int(11) default null,
primary key (id),key cl_notice_user(user_id),
constraint cl_notice_user foreign key(user_id) references user_inf(id))
engine=innodb auto_increment=19 default charset=utf8;


//创建document_inf表
create table document_inf(id int(11) not null auto_increment,title varchar(50) not null,
filename varchar(300) not null,remark varchar(300) default null,
create_date timestamp not null default current_timestamp,user_id int (11) default null,
primary key(id), key cl_document_user(user_id),
constraint cl_document_user foreign key (user_id) references user_inf(id))engine=innodb
auto_increment=7 default charset=utf8;
